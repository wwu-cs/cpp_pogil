  \Q If the parentheses were removed from the C++ Expression in the second row
    so that it was {\tt 3+5/2}, what would its value be?
    \begin{answer}[0.5in]
      The new value would be $3+\frac{5}{2} = 3 + 2 = 5$
    \end{answer}

  \Q What is stored in memory after each assignment statement below is executed?  Assume
    all variables have been declared as integers.
    \begin{center}
      \renewcommand{\arraystretch}{1.5}

        \hline

        \hline

  \Q Convert each expression to a C++ arithmetic expression.  Do not compute
    its value.
    \begin{enumerate}
      \itemsep 10pt
      \item $\displaystyle 3(6-1)$             \hfill \ans[4in]{{\tt 3*(6-1)}}
      \item $\displaystyle 12-\frac{3+2}{6-1}$ \hfill \ans[4in]{{\tt 12-(3+2)/(6-1)}}\\[-10pt]
      \item $\displaystyle \frac{3+2^2}{5}$    \hfill \ans[4in]{{\tt (3+2*2)/5}}
      \item Eight to the fourth power          \hfill \ans[4in]{{\tt 8*8*8*8}}
    \end{enumerate}

  \Q What is the value of each C++ arithmetic expression below?\key\\[-2.5mm]
    \par\vskip 10pt      
    \begin{enumerate}
      \itemsep 10pt
      \item {\tt 4 + 3 \% 2}          \hfill \ans[4in]{{\tt 4 + 3 \% 2 = 4 + 1 = 5}}
      \item {\tt 3 + 9 \% 2 * (-1+3)} \hfill \ans[4in]{{\tt 3 + 1 * 2 = 3 + 2 = 5}}\\[-10pt]
