    \Q In your own words, describe what each of the following
      vector operations does.
      \par\vskip 10pt
      
      \begin{enumerate}[(a)]
        \item \cpp{myVec.resize(n)}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Changes the size of the vector to {\tt n}.
          \end{answer}
        \item \cpp{myVec.push_back(value)}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Adds a value to the end of the vector, resizing it if
            necessary.
          \end{answer}
        \item \cpp{myVec.pop_back()}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Removes a value from the end of the vector (and returns
            the value).
          \end{answer}
        \item \cpp{myVec.begin()}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Points to the beginning of the vector.
          \end{answer}
        \item \cpp{myVec.insert(location,value)}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Inserts a value at the given location, moving everything
            over to make room.
          \end{answer}
        \item \cpp{myVec.end()}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Identifies the end of the vector (one slot past the end
            actually).
          \end{answer}
        \item \cpp{myVec.erase(location)}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Removes a value from the given location, shifting later
            values left.
          \end{answer}
        \item \cpp{myVec.clear()}%\ifprintanswers\vskip -8pt\fi
          \begin{answer}[0.5in]
            Removes all values from the vector and sets its size to zero.
          \end{answer}
      \end{enumerate}
