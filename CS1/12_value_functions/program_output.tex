    \Q Now write the function definition assuming that the function header
      is as shown below.

        \cpp{string getMessage(int userNum, int compNum)}

      \begin{answer}[2.25in]

string getMessage(int userNum,int compNum) {
  string message;
  if (userNum < compNum) {
    message = "I'm sorry, your number is smaller.\n";
  } else if (userNum > compNum) {
    message = "I'm sorry, your number is larger.\n";
  } else {
    message = "Congratulations!  You picked the number.\n";
  }
  return message;
}
