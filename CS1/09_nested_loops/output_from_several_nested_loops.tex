  \Q There are several ways to write a C++ program that outputs
    the contents of box (a) above.  Briefly describe how it could be
    done using the following methods.
    
    \begin{enumerate}
      \itemsep 15pt
      \item Using a set of {\tt cout} statements without any loops.
        \begin{answer}[0.75in]
          \par
          Create seven identical {\tt cout} statements,
          each of which prints {\tt * * * * *} and an {\tt endl}.
        \end{answer}

      \item Using a single \cpp{for} loop (without any nesting).
        \begin{answer}[0.75in]
          \par
          The \cpp{for} loop would execute seven times
          and each time {\tt cout} the five {\tt *}'s and an {\tt endl}.
        \end{answer}
    \end{enumerate}
    
  \Q In this exercise we will create the output in box (a) using nested loops.    
    \itemsep 10pt
    \begin{enumerate}
      \item Surround the {\tt cout} statement below by a \cpp{for}
        so that five {\tt *} characters are printed on a single line.
        \vskip 5pt
        \begin{minipage}{2.5in}
          \begin{center}
            \cpp{cout << "* ";}
          \end{center}
        \end{minipage}
        \begin{minipage}{3in}
          \begin{answer}[0.5in]
            \begin{center}
              \scriptsize
              \begin{cpplst}
for (int i=0; i<5; i++) {
  cout << "* ";
}
              \end{cpplst}
          \end{center}
        \end{answer}
        \end{minipage}

      \item Now build on the block of code above by wrapping the
        loop you created in another \cpp{for} loop that
        prints seven of these lines, with an {\tt endl} between each one.
        \begin{answer}[1.25in]
          \begin{center}
            \scriptsize
            \begin{minipage}{2.5in}
              \begin{cpplst}
for (int j=0; j<7; j++) {                
  for (int i=0; i<5; i++) {
    cout << "* ";
  }
  cout << endl;
}
              \end{cpplst}
            \end{minipage}
          \end{center}
        \end{answer}
    \end{enumerate}

  \vskip -30pt
      
  \Q Suppose that variables \cpp{int rows} and
    \cpp{int cols} contain the number of rows and \key\\[-2.5mm]
    columns of {\tt *}'s you wish to print.  How
    would you modify your code above to print out that number of
    rows and columns?  Test your solution in {\tt activity09b.cpp}.
    
    \begin{answer}[1.25in]
      \begin{center}
        \scriptsize
        \begin{minipage}{2.5in}
          \begin{cpplst}
for (int j=0; j<rows; j++) {
  for (int i=0; i<cols; i++) {
    cout << "* ";
  }
  cout << endl;
}
          \end{cpplst}
        \end{minipage}
      \end{center}
    \end{answer}
