    \Q There are several ways to write a C++ program that outputs
      the contents of box (a) above.  Briefly describe how it could be
      done using the following methods.
      \par\vskip 15pt
      
      \begin{enumerate}[(a)]
        \item Using a set of {\tt cout} statements without any loops.
          \begin{answer}[0.75in]
            \par
            Create seven identical {\tt cout} statements,
            each of which prints {\tt * * * * *} and an {\tt endl}.
          \end{answer}
        \item Using a single \cpp{for} loop (without any nesting).
          \begin{answer}[0.75in]
            \par
            The \cpp{for} loop would execute seven times
            and each time {\tt cout} the five {\tt *}'s and an {\tt endl}.
          \end{answer}
      \end{enumerate}
      
    \Q In this exercise we will create the output in box (a) using nested loops.    
      \par\vskip 10pt
      
      \begin{enumerate}[(a)]
        \item Surround the {\tt cout} statement below by a \cpp{for}
          so that five {\tt *} characters are printed on a single line.
          \ifprintanswers
            \begin{answer}
              \scriptsize\vskip -35pt\null
              \begin{center}
                \begin{minipage}{2.5in}
                  \begin{cprlst}[
                    frame=lines,
                    framesep=2mm,
                    bgcolor=gray!15,
                    baselinestretch=1.2,
                  ]{cpp}
for (int i=0; i<5; i++) {
  cout << "* ";
}
                  \end{cprlst}
                \end{minipage}
              \end{center}\vskip -20pt\null
            \end{answer}
          \else
            \par\vskip 10pt\null
            \begin{center}
              \mintinline{cpp}{cout << "* ";}
            \end{center}
            \par\vskip 10pt\null
          \fi
        \item Now build on the block of code above by wrapping the
          loop you created in another \cpp{for} loop that
          prints seven of these lines, with an {\tt endl} between each one.
          \begin{answer}[1.25in]
            \scriptsize\vskip -35pt\null
            \begin{center}
              \begin{minipage}{2.5in}
                \begin{cprlst}[
                  frame=lines,
                  framesep=2mm,
                  bgcolor=gray!15,
                  baselinestretch=1.2,
                ]{cpp}
for (int j=0; j<7; j++) {                
  for (int i=0; i<5; i++) {
    cout << "* ";
  }
  cout << endl;
}
                \end{cprlst}
              \end{minipage}
            \end{center}\vskip -20pt\null
          \end{answer}
      \end{enumerate}
